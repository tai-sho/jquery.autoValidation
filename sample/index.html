jQuery AutoValidation Samples

      
$(function() {
  $('#validation').autoValidation({
    '#username' : {
      'required' : null,
      'alphanumeric' : {'msg' : '半角英数字で入力！'}
    },
    '#age' : {
      'number' : {'pitch' : 'half'}
    },
    '#email' : {
      'email' : null
    },
    '#password' : {
      'required' : null,
      'between' : {'min' : 6, 'max' : 12},
      'alphanumeric' : null
    },
    '#password-conf' : {
      'required' : null,
      'equals' : {'target' : '#password', 'name' : 'password'}
    },
    '#pattern' : {
      'pattern' : {'reg' : '^[a-zA-Z0-9]+$'}
    },
    '#my-rule' : {
      'myrule1' : {'rule' : function(str) { return (str === 'cat' || str === 'dog'); }}
    }
  });
});
